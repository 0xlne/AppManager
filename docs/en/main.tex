% -------------------

\newenvironment{amalert}[1]
{\mdframed[linecolor={#1},backgroundcolor=white,linewidth=3pt,topline=false,rightline=false,bottomline=false]}
{\endmdframed}

\newenvironment{tip}[1][Tip]
{%
    \amalert{black}
    \noindent
    \textbf{\textit{#1.}}
    }%
    {
    \endamalert
}

\newenvironment{warning}[1][Warning]
{
    \amalert{Dandelion}
    \noindent
    \textbf{\textit{#1.}}
    }%
    {
    \endamalert
}

\newenvironment{danger}[1][Danger]
{
    \amalert{BrickRed}
    \noindent
    \textbf{\textit{#1.}}
    }%
    {
    \endamalert
}
